implementation 'com.zhxh:xcomponentlib:2.8'

                android:background="@color/color_white"
                android:overScrollMode="never"
                android:scrollbars="none" />

        </com.zhxh.xcomponentlib.xstickyhorizon.XStickyNavContainer>
